package firstDay;

public class FirstClass
 {
    public static void main(String[] args) {

        try {
            int a = 14;
            int b = 0;
            int result = a / b;
            System.out.println(result);
        }
        catch (ArithmeticException e) {
            System.out.println("Cannot divide by zero");
        }
        finally {
            System.out.println("Program ended");
        }
    }
}
